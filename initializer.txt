npm i -g pnpm
npx create-turbo@latest

pnpm install
pnpm run dev

inside backend folders in apps:
    npm init -y        
    import typescript config from packages in both backend apps
    add the same package as devDependency in packages.json of backend apps
    "devDependencies": {
        "@repo/typescript-config": "workspace:*"
    },
    
    run pnpm install globally(root project folder)


add build, dev and start scripts in backend servers packages.json:
"scripts": {
    "build": "tsc -b",
    "start": "node dist/index.js",
    "dev": "npm run build && npm run start"
},

add src and destination folders for both backend apps in tsconfig.json:
"compilerOptions": {
    "rootDir": "./src",
    "outDir": "./dist"
}



adding express dependency inside backend servers: 
    pnpm add express @types/express


adding another package(backend-common) in packages:
    npm init -y
    change name in package.json
    do tsconfig.json
    add tsconfig as devDependencies in package.json
    in backend-common, pnpm add @types/node  for process.env
    then add export options in package.json



adding db package:
    create db folder in packages
    npm init-y
    create tsconfig.json and make it extend, then add as devDependencies in package.json
    then global pnpm install

    create src/index.tsc
    pnpm install prisma (inside db)
    npx prisma init (inside db)

    do migrating(inside db package):
        npx prisma migrate dev --name init_schema
        pnpm i @prisma/client
        npx prisma generate
    update package.json to export the db package
